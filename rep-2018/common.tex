\tikzgraphsset{
	declare={ReactiveSequence}{
		to/\(\rightarrow\)[
			regular polygon,
			regular polygon sides = 4,
			inner sep = 0,
			minimum size = 2.5em,
			draw
		]
	}
}

			draw
		]
	}
}
